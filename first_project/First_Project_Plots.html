Function to rebin data¶

In [4]:
def rebin(matrix, xbin, ybin):

    shape1=matrix.shape[0]//xbin
    shape2=matrix.shape[1]//ybin
    return np.nanmean(np.reshape(matrix[:shape1*xbin,:shape2*ybin],(shape1,xbin,shape2,ybin)),axis=(1,3))
